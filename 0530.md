```python
my_list = [1, 2, 3, 4]

# 下标i处插入
my_list.insert(2, 'x')
print(my_list)  # 输出: [1, 2, 'x', 3, 4]
# 删除下标i的元素，并返回
removed = my_list.pop(1)
print(removed)  # 输出: 2
print(my_list)  # 输出: [1, 3, 4]
my_list.pop() # 不指定索引则删除最后一个元素
print(my_list)  # 输出: [1, 3]
# 删除指定怨怒是
my_list.remove(2)
print(my_list)  # 输出: [1, 3, 2, 4]

```
